def add_to_cart(request, product_id):
    if not request.user.is_authenticated:
        messages.error(request, '–ù—É–∂–Ω–æ –≤–æ–π—Ç–∏ –≤ –∞–∫–∫–∞—É–Ω—Ç —á—Ç–æ–±—ã –¥–æ–±–∞–≤–∏—Ç—å –≤ –∫–æ—Ä–∑–∏–Ω—É!')
        return redirect('login')
    
    product = get_object_or_404(TeaProduct, id=product_id)
    cart_item, created = Cart.objects.get_or_create(
        user=request.user,
        product=product,
        defaults={'quantity': 1}
    )
    
    if not created:
        cart_item.quantity += 1
        cart_item.save()
        messages.success(request, f'–î–æ–±–∞–≤–ª–µ–Ω –µ—â—ë –æ–¥–∏–Ω "{product.name}" –≤ –∫–æ—Ä–∑–∏–Ω—É!')

    return redirect('tea_list')

def cart_view(request):
    if not request.user.is_authenticated:
        messages.error(request, '–í–æ–π–¥–∏—Ç–µ –≤ –∞–∫–∫–∞—É–Ω—Ç —á—Ç–æ–±—ã –ø–æ—Å–º–æ—Ç—Ä–µ—Ç—å –∫–æ—Ä–∑–∏–Ω—É')

        'cart_items': cart_items,

def remove_from_cart(request, item_id):
    if not request.user.is_authenticated:
        return redirect('login')
    
    cart_item = get_object_or_404(Cart, id=item_id, user=request.user)
    product_name = cart_item.product.name

    return redirect('cart_view')

def update_cart(request, item_id):
    if not request.user.is_authenticated:
        return redirect('login')
    
    cart_item = get_object_or_404(Cart, id=item_id, user=request.user)
    new_quantity = request.POST.get('quantity')
    
    if new_quantity and new_quantity.isdigit():
        cart_item.quantity = int(new_quantity)
        if cart_item.quantity > 0:
            cart_item.save()
            messages.success(request, '–ö–æ–ª–∏—á–µ—Å—Ç–≤–æ –æ–±–Ω–æ–≤–ª–µ–Ω–æ!')
